#!/bin/bash -e

configure_disk() {
    # D945GCLF2 does not support GPT partition tables ...

    message "Partitioning"
    echo -e "size=256MiB type=83 start=1MiB bootable\nsize=16Gib type=83\ntype=83" | sfdisk /dev/sda --unit S

    message "Formatting"
    mkfs.ext4 -L boot /dev/sda1
    mkfs.ext4 -L root /dev/sda2
    mkfs.ext4 -L home /dev/sda3

    message "Mounting"
    mount LABEL=root /mnt       ; mkdir -p /mnt/{boot,home}
    mount LABEL=boot /mnt/boot
    mount LABEL=home /mnt/home
}
export -f configure_disk

configure_network() {
    lines "/etc/udev/rules.d/10-network.rules" \
        "SUBSYSTEM==\"net\", ACTION==\"add\", ATTR{address}==\"00:27:0e:01:9c:0c\", NAME=\"net0\""

    lines "/etc/netctl/net0" \
        "Interface=net0" \
        "Connection=ethernet" \
        "IP=dhcp"

    netctl enable net0
}
export -f configure_network

export INSTALL_HOST="${1:-j0057.github.io/ansible}"
export INSTALL_HOSTNAME="muon"
export INSTALL_TZNAME="Europe/Amsterdam"
export INSTALL_LOCALE="en_US.UTF-8 UTF-8"
export INSTALL_PACKAGES="base grub lsb-release python2 openssh git base-devel"
export INSTALL_SERVICES="sshd.service"

source <(curl -s "http://${INSTALL_HOST}/arch_stage0.txt")
